macro_rules! vrq {
    ( ) => { ... };
    ( $( $attr:expr ),* ) => { ... };
}
Expand description

Creates a “Voucher Request” instance with a known list of attributes.

Examples

use minerva_voucher::{Voucher, attr::*, vrq};

let v = vrq![
    Attr::Assertion(Assertion::Proximity),
    Attr::SerialNumber(b"00-11-22-33-44-55".to_vec())];

assert!(v.is_vrq());
assert_eq!(v.len(), 2);
